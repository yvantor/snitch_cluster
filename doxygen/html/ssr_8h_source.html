136static inline uint32_t read_ssr_cfg(const snrt_ssr_reg_t reg,
137 const snrt_ssr_dm_t dm) {
138 uint32_t value;
139 asm volatile("scfgri %[value], %[dm] | %[reg]<<5\n"
140 : [ value ] "=r"(value)
141 : [ dm ] "i"(dm), [ reg ] "i"(reg));
142 return value;
143}
