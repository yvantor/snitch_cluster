31static inline uint32_t snrt_dma_start_1d(uint64_t dst, uint64_t src,
32 size_t size,
33 const uint32_t channel = 0) {
34 uint32_t dst_lo = dst & 0xFFFFFFFF;
35 uint32_t dst_hi = dst >> 32;
36 uint32_t src_lo = src & 0xFFFFFFFF;
37 uint32_t src_hi = src >> 32;
38 uint32_t txid;
39
40 asm volatile(
41 "dmsrc %[src_lo], %[src_hi] \n"
42 "dmdst %[dst_lo], %[dst_hi] \n"
43 "dmcpyi %[txid], %[size], (%[channel] << 2) | 0b00 \n"
44 : [ txid ] "=r"(txid)
45 : [ src_lo ] "r"(src_lo), [ src_hi ] "r"(src_hi),
46 [ dst_lo ] "r"(dst_lo), [ dst_hi ] "r"(dst_hi), [ size ] "r"(size),
47 [ channel ] "i"(channel));
48
49 return txid;
50}

113static inline snrt_dma_txid_t snrt_dma_start_2d(uint64_t dst, uint64_t src,
114 size_t size, size_t dst_stride,
115 size_t src_stride,
116 size_t repeat,
117 const uint32_t channel = 0) {
118 uint32_t dst_lo = dst & 0xFFFFFFFF;
119 uint32_t dst_hi = dst >> 32;
120 uint32_t src_lo = src & 0xFFFFFFFF;
121 uint32_t src_hi = src >> 32;
122 uint32_t txid;
123
124 asm volatile(
125 "dmsrc %[src_lo], %[src_hi] \n"
126 "dmdst %[dst_lo], %[dst_hi] \n"
127 "dmstr %[src_stride], %[dst_stride] \n"
128 "dmrep %[repeat] \n"
129 "dmcpyi %[txid], %[size], (%[channel] << 2) | 0b10 \n"
130 : [ txid ] "=r"(txid)
131 : [ src_lo ] "r"(src_lo), [ src_hi ] "r"(src_hi),
132 [ dst_lo ] "r"(dst_lo), [ dst_hi ] "r"(dst_hi),
133 [ dst_stride ] "r"(dst_stride), [ src_stride ] "r"(src_stride),
134 [ repeat ] "r"(repeat), [ size ] "r"(size), [ channel ] "i"(channel));
135
136 return txid;
137}

165static inline void snrt_dma_wait(snrt_dma_txid_t txid,
166 const uint32_t channel = 0) {
167 asm volatile(
168 "1: \n"
169 "dmstati t0, (%[channel] << 2) | 0 \n"
170 "bltu t0, %[txid], 1b \n"
171 :
172 : [ txid ] "r"(txid), [ channel ] "i"(channel)
173 : "t0");
174}

185static inline void snrt_dma_wait_all(const uint32_t channel = 0) {
186 uint32_t busy;
187 asm volatile(
188 "1: \n"
189 "dmstati %[busy], (%[channel] << 2) | 2 \n"
190 "bne %[busy], zero, 1b \n"
191 : [ busy ] "=r"(busy)
192 : [ channel ] "i"(channel));
193}

227inline void snrt_dma_memset(void *ptr, uint8_t value, uint32_t len) {
228 // set first 64bytes to value
229 // memset(ptr, value, 64);
230 uint8_t *p = (uint8_t *)ptr;
231 uint32_t nbytes = 64;
232 while (nbytes--) {
233 *p++ = value;
234 }
235
236 // DMA copy the the rest
237 snrt_dma_txid_t memset_txid =
238 snrt_dma_start_2d(ptr, ptr, 64, 64, 0, len / 64);
239 snrt_dma_wait_all();
240}

340inline snrt_dma_txid_t snrt_dma_load_2d_tile(
341 void *dst, void *src, size_t tile_x1_idx, size_t tile_x0_idx,
342 size_t tile_x1_size, size_t tile_x0_size, size_t full_x0_size,
343 uint32_t prec, size_t tile_ld) {
344 size_t src_offset = 0;
345 // Advance src array in x0 and x1 dimensions, and convert to byte offset
346 src_offset += tile_x0_idx * tile_x0_size;
347 src_offset += tile_x1_idx * tile_x1_size * full_x0_size;
348 src_offset *= prec;
349 // Initiate transfer
350 return snrt_dma_start_2d((uint64_t)dst, // dst
351 (uint64_t)src + src_offset, // src
352 tile_x0_size * prec, // size
353 tile_ld, // dst_stride
354 full_x0_size * prec, // src_stride
355 tile_x1_size // repeat
356 );
357}
